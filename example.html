1. 调用方式

/*最简单调用方式*/
 $(".postcontent").html580Ads();
/*参数调用*/ 
  $(".postcontent").html580Ads({
  	width: 468,              //广告位宽度
    height: 60,              //广告位高度
    minHeight:200,           //图片显示广告最小高度
    opacity:'0.9',           //透明度
    offsetHeight:0,          //广告位相对图片的高度位移量
    offsetWidth:0,           //广告位相对图片的宽度位移量
    adsDom:'#html580_ads',   //广告位结点
    initShow:true            //是否显示
  });
